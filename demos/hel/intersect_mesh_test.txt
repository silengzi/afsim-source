# HEL and intersection mesh test.
#
# BLUE_MULTIROLE_FIGHTER_1EF W/HEL engages an air target
# A HEL-equipped blue player chases a red
# player, intending to shoot the IRST (default
# region on the mesh).
# The script continually checks the
# target to see if the IRST is occluded.
# Initially it is occluded. so
# there is no shot until the red player maneuvers
# and the IRST becomes visible.
#
#---------------------------------------------
define_path_variable CASE intersect_mesh_test
log_file output/$(CASE).log

weapon_effects hel_lethality WSF_HEL_LETHALITY

   region RED_ADV_FIGHTER_2 CANOPY
      minimum_energy 5 kj
   end_region

   region RED_ADV_FIGHTER_2 IRST
      minimum_energy 2 kj
      damage_radius 5 cm
   end_region

   #debug
end_weapon_effects

processor auto_fire_computer WSF_SCRIPT_PROCESSOR
   on
   update_interval 1.0 s

   script_variables
      double fire_duration = 10.0;
      double fire_complete_time = 0.0;
      string weapon_name = "hel";
      WsfTrack mActiveTrack;
      WsfLaserWeapon weapon_system;
      double minimum_range = 20000.0; # 10 km
      double minimum_target_speed = 350.0; #350 m/s
   end_script_variables

   on_initialize
      weapon_system = (WsfLaserWeapon)PLATFORM.Weapon(weapon_name);
   end_on_initialize

   # Logic to determine if a track is a missile
   script bool IsOK(WsfTrack aTrack)
      WsfPlatform targetPlatform = WsfSimulation.FindPlatform(aTrack.TargetIndex());
      bool isOk = targetPlatform.IsValid() && (targetPlatform.Side() != PLATFORM.Side());
      return isOk;
   end_script

   script bool IsOccluded()
      bool occluded = true;
      if (mActiveTrack.IsValid())
      {
         string offset = mActiveTrack.AuxDataString("TARGET_OFFSET");
         if (WsfSimulation.FindPlatform(mActiveTrack.TargetIndex()).IsValid())
         {
            occluded = weapon_system.TargetOccluded(mActiveTrack.TargetName(), offset);
            if (occluded)
            {
               writeln(mActiveTrack.TargetName(), ": ", offset, " occluded.");
            }
            else
            {
               writeln(mActiveTrack.TargetName(), ": ", offset, " not occluded.");
            }
         }
      }
      return occluded;
   end_script


   script WsfTrack FindBestTarget()
      double bestDistance = 1.0E20;
      WsfTrack missileTrack = null;
      WsfLocalTrackList tracks = PLATFORM.MasterTrackList();
      for (int i = 0; i < tracks.Count(); i=i+1)
      {
         WsfLocalTrack track = tracks.Entry(i);
         //writeln("Examining track, name ", track.TargetName());
         if (IsOK(track))
         {
            double dist = track.SlantRangeTo(PLATFORM);
            if (dist < bestDistance && dist < minimum_range)
            {
               bestDistance = dist;
               missileTrack = track;
            }
         }
      }
      if (missileTrack.IsValid())
      {
         writeln("found ", missileTrack.TargetName());
      }
      return missileTrack;
   end_script

   on_update
      if (!weapon_system.IsTurnedOn())
      {
         writeln("Not turned on...");
         return;
      }
      if (!weapon_system.IsValid())
      {
         writeln("Not valid...");
         return;
      }

      if (fire_complete_time > TIME_NOW)
      {
         if (mActiveTrack.IsValid())
         {
            string offset = mActiveTrack.AuxDataString("TARGET_OFFSET");
            if (! WsfSimulation.FindPlatform(mActiveTrack.TargetIndex()).IsValid())
            {
               fire_complete_time = TIME_NOW;
               weapon_system.CeaseFire();
               writeln("...Target ", mActiveTrack.TargetName(), " destroyed...");
            }
            else if (IsOccluded())
            {
               fire_complete_time = TIME_NOW;
               weapon_system.CeaseFire();
               writeln("...Target ", mActiveTrack.TargetName(), ":", offset, " occluded...");
            }
         }
         return;
      }
      if (weapon_system.IsFiring())
      {
         writeln(TIME_NOW, " ", PLATFORM.Name(), ": (cease fire).");
         weapon_system.CeaseFire();
      }
      WsfTrack target = FindBestTarget();
      if (target.IsValid())
      {
         mActiveTrack = target;
         //string offset = "CANOPY";
         string offset = "IRST";
         mActiveTrack.SetAuxData("TARGET_OFFSET", offset);
         if (! IsOccluded())
         {
            writeln(TIME_NOW, ": firing at target: ", mActiveTrack.TargetName(), offset);
            weapon_system.Fire(mActiveTrack.TargetName(), offset);
            fire_complete_time = TIME_NOW + fire_duration;
         }
      }
      else
      {
         //writeln("Target not valid!");
      }
   end_on_update

end_processor

weapon HEL WSF_LASER_WEAPON

   aux_data
      bool auto_mode = false
   end_aux_data

   #debug
   comments enabled

   // The firing time per shot.
   firing_time 5.0 sec

   // The lethalitiy model used by this HEL.
   weapon_effects hel_lethality

   //-------------------------------------------------------------------
   //  The following begins the hel-specific block of data.
   //-------------------------------------------------------------------

   fluence_model default

      // The aperture for the HEL beam.
      // The beam diffraction is inversely proportional to the aperture diameter.
      aperture_diameter 0.2 meters

      // The laser power (typically, expressed in kw).
      power 20 kw

      // The characteristic wavelength for the measurement.
      //wavelength 1.064 microns

      wavelength 1064e-9 meter
      //atmosphere_model 1
      //haze_model 1

      // The total jitter of the beam on target.
      // More jitter gives a larger spot size and less power/area.
      jitter 2e-6 radians

      // A measure of how close the beam is to a perfect gaussian.
      // Valid values are >= 1.0, where 1.0 is best.
      beam_quality 1.2

   end_fluence_model

end_weapon

platform red_adv_fighter_2 RED_ADV_FIGHTER_2
   add processor turn-proc WSF_SCRIPT_PROCESSOR
      script void turn()
         static double heading = 180.0;
         if (heading == 180.0)
         {
            heading = 0.0;
         }
         else
         {
            heading = 180.0;
         }
         PLATFORM.TurnToHeading(heading);
      end_script
      on_update
         turn();
      end_on_update
      update_interval 30.0 seconds
   end_processor

   route
      position 0:10:00n 0.0383e speed 200 kts altitude 2000 meters heading 180 degrees
   end_route
