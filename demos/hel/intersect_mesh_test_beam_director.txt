processor auto_fire_computer WSF_SCRIPT_PROCESSOR
   on
   update_interval 1.0 s

   script_variables
      double fire_duration = 10.0;
      double fire_complete_time = 0.0;
      string weapon_name = "hel";
      bool hasTarget = false;
      WsfTrack mActiveTrack = WsfTrack();
      WsfLaserWeapon weapon_system;
      double minimum_range = 20000.0; // 20 km
      double minimum_target_speed = 350.0; // 350 m/s
   end_script_variables

   on_initialize
      weapon_system = (WsfLaserWeapon)PLATFORM.Weapon(weapon_name);
   end_on_initialize

   // Logic to determine if a track is a missile
   script bool IsOK(WsfTrack aTrack)
      WsfPlatform targetPlatform = WsfSimulation.FindPlatform(aTrack.TargetIndex());
      bool isOk = targetPlatform.IsValid() && (targetPlatform.Side() != PLATFORM.Side());
      return isOk;
   end_script

   script bool IsOccluded()
      bool occluded = true;
      if (mActiveTrack.IsValid())
      {
         string offset = mActiveTrack.AuxDataString("TARGET_OFFSET");
         if (WsfSimulation.FindPlatform(mActiveTrack.TargetIndex()).IsValid())
         {
            occluded = weapon_system.TargetOccluded(mActiveTrack.TargetName(), offset);
            if (occluded)
            {
               writeln(TIME_NOW, "; ", mActiveTrack.TargetName(), ": ", offset, " occluded.");
            }
            else
            {
               writeln(TIME_NOW, "; ", mActiveTrack.TargetName(), ": ", offset, " not occluded.");
            }
         }
      }
      return occluded;
   end_script

   script WsfTrack FindBestTarget()
      double bestDistance = 1.0E20;
      WsfTrack missileTrack = null;
      WsfLocalTrackList tracks = PLATFORM.MasterTrackList();
      for (int i = 0; i < tracks.Count(); i=i+1)
      {
         WsfLocalTrack track = tracks.Entry(i);
         #writeln("Examining track, name ", track.TargetName());
         if (IsOK(track))
         {
            double dist = track.SlantRangeTo(PLATFORM);
            if (dist < bestDistance && dist < minimum_range)
            {
               bestDistance = dist;
               missileTrack = track;
            }
         }
      }
      if (missileTrack.IsValid())
      {
         writeln("found ", missileTrack.TargetName());
      }
      return missileTrack;
   end_script

   on_update
      if (!weapon_system.IsTurnedOn())
      {
         writeln("Not turned on...");
         return;
      }
      if (!weapon_system.IsValid())
      {
         writeln("Not valid...");
         return;
      }

      if (weapon_system.IsFiring())
      {
         hasTarget = false;
         return;
      }
      else
      {
         WsfTrack target = FindBestTarget();
         if (target.IsValid() &&
            (! hasTarget))
         {
            writeln("Acquiring target ", target.TargetName());
            mActiveTrack = target;
            #string offset = "CANOPY";
            string offset = "IRST";
            writeln(TIME_NOW, ": firing at target: ", mActiveTrack.TargetName(), offset);

            hasTarget = weapon_system.Fire(mActiveTrack.TargetName(), offset);
            #hasTarget = weapon_system.Fire(mActiveTrack, offset);
         }

        IsOccluded();
      }

      if (mActiveTrack.IsValid())
      {
         string offset = mActiveTrack.AuxDataString("TARGET_OFFSET");
         if (! WsfSimulation.FindPlatform(mActiveTrack.TargetIndex()).IsValid())
         {
            weapon_system.CeaseFire();
            hasTarget = false;
         }
      }

   end_on_update

end_processor

weapon HEL WSF_CUED_LASER_WEAPON

   aux_data
      bool auto_mode = false
   end_aux_data

   #debug
   comments enabled

   // The firing time per shot.
   firing_time 5.0 sec

   // The lethalitiy model used by this HEL.
   weapon_effects hel_lethality

   //-------------------------------------------------------------------
   //  The following begins the hel-specific block of data.
   //-------------------------------------------------------------------

   fluence_model default

      // base_input_file ./intersect_test.hel

      // helcomes_install_folder C:\pkgs\helcomes\helcomes

      // helcomes_interface_jar c:\swdev\cme\wsf_helcomes\lib\wsf.jar
      // thermal_bloom enabled

      // The aperture for the HEL beam.
      // The beam diffraction is inversely proportional to the aperture diameter.
      aperture_diameter 0.2 meters

      // The laser power (typically, expressed in kw).
      power 20 kw

      // The characteristic wavelength for the measurement.
#      wavelength 1.064 microns

#      wavelength 1064e-9 meter
      laser_type nd_yag

#      atmosphere_model 1
#      haze_model 1

      // The total jitter of the beam on target.
      // More jitter gives a larger spot size and less power/area.
      jitter 2e-6 radians

      // A measure of how close the beam is to a perfect gaussian.
      // Valid values are >= 1.0, where 1.0 is best.
      beam_quality 1.2

   end_fluence_model

end_weapon

        location                0 0 0 m  // tbd

          frame_time                    1 sec
          revisit_time                0.1 sec

          // fovs based on scan volume

          elevation_field_of_view -4.0 deg 4.0 deg

          // the errors are the same as the beam diveregence (or less)
#             azimuth_error_sigma   0.0 deg
#             elevation_error_sigma 0.0 deg
#             range_error_sigma 0.0 m

          // Track Formation Commands
          reports_bearing
          reports_elevation
          reports_location
          reports_velocity

        end_mode_template

         // Following are the sensor modes.  When the beam director is cued to a track, it begins in the
         // first mode.  The on_success and on_failure inputs indicate mode transitions.  In the example
         // below there are four modes.  They progress from one to the next in order.  If track is lost in
         // one mode, it falls back to the prior mode.
        mode SCAN
          // Search to acquire tracks
          scan_mode                 azimuth_and_elevation
          frame_time                0.1 s
          azimuth_scan_limits       -4.0 degrees 4.0 degrees
          elevation_scan_limits     -4.0 degrees 4.0 degrees

          // If maximum_request_count > 0 then this mode only responds
          //   to explicit requests as initiated by WsfTaskManager::StartTracking.
          track_quality             0.8  // Near 'weapon quality'
          hits_to_establish_track   1 1  // 1 of last 1 scans to establish track
          hits_to_maintain_track    1 5  // 1 of last 5 scans to maintain  track
          type                      scanning
          on_success                LOCK1
        end_mode

        mode LOCK1
          scan_mode                 fixed
          frame_time                5.0 s  // Implement scan mode delay here; it must be
                                           // enforced in one of the fixed modes
          track_quality             0.9  // 'weapon quality'
          hits_to_establish_track   1 1  // 1 of last 1 scans to establish track
          hits_to_maintain_track    1 5  // Allow 25 seconds of coasting (5th failure will drop)
          type                      locked
          on_success                LOCK2
          on_failure                SCAN
        end_mode

        mode LOCK2
          scan_mode                 fixed
          frame_time                0.1 s
          track_quality             1.0  // 'weapon quality'
          hits_to_establish_track   1 1  // 1 of last 1 scans to establish track
          hits_to_maintain_track    1 5  // Allow .5 seconds of coasting (5th failure will drop)
          type                      locked
          on_success                CUED
          on_failure                LOCK1
        end_mode

        mode CUED
          scan_mode                 fixed
          frame_time                0.01 s
          track_quality             1.0  // 'weapon quality'
          hits_to_establish_track   1 1  // 1 of last 1 scans to establish track
          hits_to_maintain_track    1 5  // Allow .05 seconds of coasting (5th failure will drop)
          on_failure                LOCK2
          type                      cued
        end_mode

   end_sensor  // beam director

end_platform_type

      script void turn()
         static double heading = 180.0;
         if (heading == 180.0)
         {
            heading = 0.0;
         }
         else
         {
            heading = 180.0;
         }
         PLATFORM.TurnToHeading(heading);
      end_script
      on_update
         turn();
      end_on_update
      update_interval 30.0 seconds
   end_processor
