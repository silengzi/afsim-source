# simple task processor to launch a sam at an air target

script_interface
#  debug
end_script_interface

# The launch computer for the weapon

script bool SAM_MISSILE_LaunchComputer(WsfTrack aTrack,
                                 WsfPlatform aLauncher,
                                 double aLaunchDelay)
   # keep in mind that all values are in meters...
   bool canIntercept = false;
   double tgtAlt = aLauncher.RelativeAltitudeOf(aTrack);
   double shootDelay = 1.0;
   double maxRng = 35000;
   double maxTOF = 692.0;
   double maxAlt = 10000;
   double avgSpd = maxRng / maxTOF;
   double maxTgtSpd = 2000;
   double tgtSpd = aTrack.Speed();
   WsfWaypoint intercept = WsfWaypoint();
   double tgtMaxHdg = 90;
   double tgtHdg = aLauncher.RelativeHeadingOf(aTrack);
   # writeln("Target speed is " , tgtSpd);
   double tti = aLauncher.InterceptLocation2D(aTrack, intercept, avgSpd, aLaunchDelay + shootDelay);
   if ((tti > 0.0) &&          # time to intercept must be above zero
       (tti < maxTOF) &&       # time to intercept must be below this
       (tgtAlt < maxAlt) &&    # target can only be so high
       (tgtSpd < maxTgtSpd) && # target can only be flying so fast
       (tgtHdg < tgtMaxHdg))   # target must be flying toward us
   {
      canIntercept = true;
   }
   return canIntercept;
end_script
# End of launch computer

# A way to add other launchers
# Calls externally to the SAM MISSILE Launch Computer
script bool SAM_LaunchComputer(WsfTrack aTrack, WsfPlatform aAssignee, double  aLaunchDelay)
   extern bool SAM_MISSILE_LaunchComputer(WsfTrack, WsfPlatform, double);
   bool canIntercept = false;
   string subType = aAssignee.Type();
   if (subType == "SAM_LAUNCHER") # PUT THE CARRYING LAUNCHER TYPE HERE
   {
      canIntercept = SAM_MISSILE_LaunchComputer(aTrack, aAssignee, aLaunchDelay);
   }

   return canIntercept;
end_script

## Can add other missiles on the launcher
script bool Local_SAM_LaunchComputer(WsfTrack aTrack, WsfWeapon aWeapon, double aLaunchDelay)
   extern bool SAM_MISSILE_LaunchComputer(WsfTrack, WsfPlatform, double);
   bool canIntercept = false;
   string weaponType = aWeapon.Type();
   WsfPlatform platform = aWeapon.Platform();
   if (weaponType == "SAM_MISSILE")
   {
      canIntercept = SAM_MISSILE_LaunchComputer(aTrack, platform, aLaunchDelay);
   }

   return canIntercept;
end_script


#---------------------------------------

processor SAM_TACTICS_PROCESSOR WSF_TASK_PROCESSOR
#   script_debug_writes on
#   show_state_transitions
   show_task_messages

   script_variables
      string SENSOR_NAME               = "radar_set";
      string SENSOR_TRACK_MODE         = "default";
      string WEAPON_NAME               = "sam_missile";
      double REQUIRED_TRACK_QUALITY    = 0.49;
      int SALVO_SIZE                   = 1;

      # the following are used internally and should not be modified by the
      # user
      string mTrackTaskStr      = "Track";
      string mShootTaskStr      = "Shoot";
   end_script_variables

   # demonstrates the use of calling an external local launch computer that could be
   # used to call other launchers to launch different missiles
   script bool InInterceptEnvelopeOf(WsfWeapon aWeapon)
      extern bool Local_SAM_LaunchComputer(WsfTrack, WsfWeapon, double);
      return Local_SAM_LaunchComputer(TRACK, aWeapon, 0.0);
   end_script


   script bool IsEngageable()
      return (((PLATFORM == PLATFORM.Commander()) ||
               (TasksReceivedFor(TRACK.TrackId()) > 0) ||
               (OperatingLevelFor("ENGAGE") > 0))) &&
               (TRACK.AirDomain());
   end_script

   script bool WeaponsAvailable()
      WsfWeapon   weapon;
      weapon = PLATFORM.Weapon(WEAPON_NAME);
      writeln_d("Weapons remaining " , weapon.QuantityRemaining());
      if (weapon.QuantityRemaining() > 0)
      {
         return true;
      }
      else
      {
         return false;
      }
      end_script

   script bool TrackingNow()
      if (SENSOR_NAME == "")
      {
         return (TRACK.TrackQuality() > REQUIRED_TRACK_QUALITY);
      }
      else
      {
         return (TRACK.TrackQuality() > 0.95);
      }
   end_script

   # returns true if weapon is still active for given track
   script bool FiringNow()
      return (WeaponsActiveFor(TRACK.TrackId()) > 0);
   end_script


   script bool LaunchWeapon()
      WsfWeapon   weapon;
      weapon = PLATFORM.Weapon(WEAPON_NAME);
      bool canInterceptNow = false;
      if (weapon.IsTurnedOn() &&
          (weapon.QuantityRemaining() >= SALVO_SIZE))
      {
         canInterceptNow = InInterceptEnvelopeOf(weapon);
      }
      bool launched = false;
      if (canInterceptNow)
      {
         weapon.CueToTarget(TRACK);
         launched = FireAt(TRACK, mShootTaskStr, weapon, SALVO_SIZE);
         if (launched)
         {
            writeln_d("*** T=", TIME_NOW, " ", PLATFORM.Name(), " ",
                      TRACK.TargetName(), " R=", PLATFORM.SlantRangeTo(TRACK),
                      " FIRE!!!!");
         }
      }
      return launched;
   end_script

#-----------------------begin state machine-----------------------
   evaluation_interval DETECTED 2.0 sec
   state DETECTED
      next_state DETECTED
         if (TRACK.IFF_Friend() || ! WeaponsAvailable())
         {
            # cancel any tasks assigned against this track
            CancelTask(TRACK.TrackId());
            return true;
         }
         return false;
      end_next_state

      next_state ENGAGE
         return (TRACK.BelievedAlive() &&
                 (! TRACK.IFF_Friend()) &&
                 IsEngageable());
      end_next_state
   end_state

   evaluation_interval ENGAGE 2.0 sec
   state ENGAGE  # Engages and fires the weapon if it can
      next_state DETECTED
         return (TRACK.BelievedDead() ||
                 ! TrackingNow());
      end_next_state

      next_state WEAPONCHECK
         bool launched = false;
         if (TRACK.BelievedAlive() &&
             TrackingNow() &&  # in case we lost the track a microsecond ago
             ! FiringNow() &&  # if a shot has been fired, wait for results
             InInterceptEnvelopeOf(PLATFORM.Weapon(WEAPON_NAME)))
         {
             writeln_d("Trying to launch weapon " );
             launched = LaunchWeapon();
         }
         return launched;
      end_next_state
   end_state

   evaluation_interval WEAPONCHECK 2.0 sec
   state WEAPONCHECK
       next_state DETECTED
           return (WeaponsAvailable());
       end_next_state

       next_state WEAPONCHECK # example of dead ending a state machine
           return true;
       end_next_state
   end_state
end_processor
