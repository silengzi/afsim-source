# This platform type is generic, and does include script variables that may not
# always be valid for all players.  For example, in this scenario, only the strike
# aircraft have need of a mechanic to repair them, but all have the
# repairOfferingPlatform variable.  For any player, any unused references are
# left undefined, so an IsValid() check should always be made prior to use.

# NOTE:  Motion functionality for players is not addressed here.  For instance,
# some players are permanently stationary, with other simulated entities coming to
# them, or the opposite, modeling the itinerant player who visits others.  It is
# up to the either the locomotive or proximity checking logic to enable or disable
# transactors which need to exchange products or services.  This is done using the
# SetEnableSupplyingFuel(), SetEnableReceivingFuel(), SetEnableSupplyingRepair(), and
# SetEnableReceivingRepair() functions.  Due to accounting problems during debugging
# and development, the transfer mechanism is made fairly rigid, and it goes like
# this:

# No transaction of a commodity or service is permitted until the previous
# transaction is known to be complete, or explicitly cancelled, and the
# other side platform has been explicitly set to NULL.  BOTH sides of the
# transaction must be known in advance, and the "other" side _MUST_ be an
# identified known supplier or receiver.   These constraints are checked
# before a new transaction can be brokered.  There are three flags used to
# orchestrate events;  xxxXxxxEnabled, xxxXxxxBegan, and xxxXxxxComplete.
# Each must begin false, then transition to true before the transaction is
# deemed complete, and another one may begin.  This makes for a ton load
# of flags, but is the only way to keep conflicts from occuring.

   script void Say(string aMsg)
      if ((runSilent || gDebugNone) && (! gDebugAll)) return;
      writeln("At t=", TIME_NOW,", ",PLATFORM.Name(), " - ", aMsg);
   end_script

   # ========== FUEL ACCOUNTING and TRANSFERS ==========

   script WsfPlatform FuelOfferingPlatform()
      return fuelOfferingPlatform;
   end_script

   script void SetFuelOfferingPlatform(WsfPlatform aPlatform)
      if (aPlatform == fuelOfferingPlatform) return;
      string oldP = "<unknown>";
      if (fuelOfferingPlatform.IsValid()) oldP = fuelOfferingPlatform.Name();
      string newP = "<unknown>";
      if (aPlatform.IsValid()) newP = aPlatform.Name();
      string msg = "Switched fuelOfferingPlatform from " + oldP + " to " + newP;
      Say(msg);
      fuelOfferingPlatform = aPlatform;
   end_script

   script WsfPlatform FuelRequestingPlatform()
      return fuelRequestingPlatform;
   end_script

   script void SetFuelRequestingPlatform(WsfPlatform aPlatform)
      if (aPlatform == fuelRequestingPlatform) return;
      string oldP = "<unknown>";
      if (fuelRequestingPlatform.IsValid())
      {
         oldP = fuelRequestingPlatform.Name();
      }
      string newP = "<unknown>";
      if (aPlatform.IsValid())
      {
         newP = aPlatform.Name();
      }
      string msg = "Switched fuelRequestingPlatform from " + oldP + " to " + newP;
      Say(msg);
      fuelRequestingPlatform = aPlatform;
   end_script

   script bool AddFuelSupplierP(WsfPlatform aPlatform)
      foreach(WsfPlatform plat in fuelPossibleSuppliers)
      {
         if (plat == aPlatform)
         {
            string msg = "Possible fuel supplier " +
               aPlatform.Name() + " was a duplicate.";
            Say(msg);
            return false;
         }
      }
      fuelPossibleSuppliers.PushBack(aPlatform);
      return true;
   end_script

   script bool AddFuelPossibleSupplier(string aPlatformName)
      WsfPlatform p = WsfSimulation.FindPlatform(aPlatformName);
      if (p.IsValid())
      {
         return AddFuelSupplierP(p);
      }
      else
      {
         string msg = "Could not add possible fuel supplier " + aPlatformName;
         Say(msg);
      }
      return false;
   end_script

   script bool AddFuelReceiverP(WsfPlatform aPlatform)
      foreach(WsfPlatform plat in fuelPossibleReceivers)
      {
         if (plat == aPlatform)
         {
            string msg = "Possible fuel receiver " +
               aPlatform.Name() + " was a duplicate.";

   end_script

   script bool AddFuelPossibleReceiver(string aPlatformName)

      {
         return AddFuelReceiverP(p);
      }

                   aPlatformName + " not found.";
      Say(msg);
      return false;

   script bool ResolvePlayersToAdd()
      bool success = true;
      foreach(string name in fuelPossibleReceiversToAdd)
      {
         if (! AddFuelPossibleReceiver(name))
         {
            string msg = "****** Couldnt AddFuelPossibleReceiver " + name + " ********";
            Say(msg);
            success = false;
         }
      }
      foreach(string name in fuelPossibleSuppliersToAdd)
      {
         if (! AddFuelPossibleSupplier(name))
         {
            string msg = "****** Couldnt AddFuelPossibleSupplier " + name + " ********";
            Say(msg);
            success = false;
         }
      }
      foreach(string name in repairPossibleReceiversToAdd)
      {
         if (! AddRepairPossibleReceiver(name))
         {
            string msg = "****** Couldnt AddRepairPossibleReceiver " + name + " ********";
            Say(msg);
            success = false;
         }
      }
      foreach(string name in repairPossibleSuppliersToAdd)
      {
         if (! AddRepairPossibleSupplier(name))
         {
            string msg = "****** Couldnt AddRepairPossibleSupplier " + name + " ********";
            Say(msg);
            success = false;
         }
      }
      return success;
   end_script

   script bool LP_Init()

      bool success = true;

      WsfProcessor p1 = PLATFORM.Processor("exchanger");
      exProc = (WsfExchangeProcessor)p1;
      if (! exProc.IsValid())
      {
         Say("WsfExchangeProcessor 'exchanger' not found.");
         success = false;
      }

      fuelContainer = exProc.FindContainer("fuel_container");
      if (! fuelContainer.IsValid())
      {
         Say("WsfExchangeProcessor 'fuel_container' not found.");
         success = false;
      }

      fuelRequestTransactor = exProc.FindTransactor("fuel_requesting_transactor");
      if (! fuelRequestTransactor.IsValid())
      {
         Say("WsfExchangeProcessor 'fuel_requesting_transactor' not found.");
         success = false;
      }

      fuelOfferTransactor = exProc.FindTransactor("fuel_offering_transactor");
      if (! fuelOfferTransactor.IsValid())
      {
         Say("WsfExchangeProcessor 'fuel_offering_transactor' not found.");
         success = false;
      }

      repairContainer = exProc.FindContainer("good_engine_container");
      if (! repairContainer.IsValid())
      {
         Say("WsfExchangeProcessor 'good_engine_container' not found.");
         success = false;
      }

      repairRequestTransactor = exProc.FindTransactor("repair_requesting_transactor");
      if (! repairRequestTransactor.IsValid())
      {
         Say("WsfExchangeProcessor 'repair_requesting_transactor' not found.");
         success = false;
      }
      repairOfferTransactor = exProc.FindTransactor("repair_offering_transactor");
      if (! repairOfferTransactor.IsValid())
      {
         Say("WsfExchangeProcessor 'repair_offering_transactor' not found.");
         success = false;
      }

      if (success)
      {
         Say("LP_Init() succeeded");
         isInitialized = true;
      }
      else
      {
         Say("LP_Init() FAILED");
      }
      return success;

   end_script
