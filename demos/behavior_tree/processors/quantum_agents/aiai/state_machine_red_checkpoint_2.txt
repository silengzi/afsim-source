advanced_behavior red_checkpoint_2
   
   script_variables
      bool reachedCheckpoint = false;
      bool commandIssued = false;
   end_script_variables

   precondition
      return Success();
   end_precondition
   
   execute
      if (reachedCheckpoint)
      {
         return Success("Checkpoint 2 reached!");
      }
      
      if (PLATFORM.GroundRangeTo(WsfSimulation.FindPlatform("Red_checkpoint_2")) < 10000)
      {
         reachedCheckpoint = true;
         return Success("Checkpoint 2 reached!");
      }
      
      if (!commandIssued)
      {
         commandIssued = true;
         FlyTarget(PLATFORM, WsfSimulation.FindPlatform("Red_checkpoint_2").Location(), 450);
      }
      return Running("Flying to Checkpoint 2");
   end_execute
   
end_advanced_behavior
