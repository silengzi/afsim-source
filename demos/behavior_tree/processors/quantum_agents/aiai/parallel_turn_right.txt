advanced_behavior turn_right
   
   script_variables
      int tick = 0;
   end_script_variables

   precondition
      return Success();
   end_precondition
   
   execute
      if (tick >= 50)
      {
         return Success("turn_right maneuver complete!");
      }
      tick = tick + 1;
      PLATFORM.TurnToRelativeHeading(5);
      return Running("Turning right");
   end_execute
   
end_advanced_behavior
