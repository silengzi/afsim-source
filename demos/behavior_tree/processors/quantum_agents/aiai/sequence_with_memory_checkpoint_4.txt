advanced_behavior checkpoint_4
   
   script_variables
      bool commandIssued = false;
   end_script_variables

   precondition
      return Success();
   end_precondition
   
   execute
      if (PLATFORM.GroundRangeTo(WsfSimulation.FindPlatform("Blue_checkpoint_4")) < 10000)
      {
         return Success("Checkpoint 4 reached!");
      }
      
      if (!commandIssued)
      {
         commandIssued = true;
         FlyTarget(PLATFORM, WsfSimulation.FindPlatform("Blue_checkpoint_4").Location(), 450);
      }
      return Running("Flying to Checkpoint 4");
   end_execute
   
end_advanced_behavior
