advanced_behavior wait
   
   script_variables
      int tick = 0;
   end_script_variables

   precondition
      return Success();
   end_precondition
   
   execute
      if (tick >= 20)
      {
         return Success("Done waiting!");
      }
      tick = tick + 1;
      return Running("Waiting");
   end_execute
   
end_advanced_behavior
