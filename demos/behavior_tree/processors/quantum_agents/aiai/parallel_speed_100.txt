advanced_behavior speed_100
   
   script_variables
      bool commandIssued = false;
      bool reachedSpeed = false;
   end_script_variables

   precondition
      return Success();
   end_precondition
   
   execute
      if (reachedSpeed)
      {
         return Success("100 m/s reached!");
      }

      if (PLATFORM.Speed() >= 99.8 && PLATFORM.Speed() <= 100.2)
      {
         reachedSpeed = true;
      }
      
      if (!commandIssued)
      {
         PLATFORM.GoToSpeed(100);
         commandIssued = true;
      }
      
      return Running("Going to 100 m/s");
   end_execute
   
end_advanced_behavior
