include_once ../common/common_platform_script.txt

advanced_behavior checkpoint_3
   
   script_variables
      bool reachedCheckpoint = false;
      bool commandIssued = false;
      double priorityValue = 3.0;
   end_script_variables

   precondition
      return priorityValue;
   end_precondition
   
   execute
      if (reachedCheckpoint)
      {
         priorityValue = 0.0;
         return Success("Checkpoint 3 reached!");
      }
      
      if (PLATFORM.GroundRangeTo(WsfSimulation.FindPlatform("Blue_checkpoint_3")) < 10000)
      {
         priorityValue = 0.0;
         reachedCheckpoint = true;
         return Success("Checkpoint 3 reached!");
      }
      
      if (!commandIssued)
      {
         commandIssued = true;
         FlyTarget(PLATFORM, WsfSimulation.FindPlatform("Blue_checkpoint_3").Location(), 450);
      }
      return Running("Flying to Checkpoint 3");
   end_execute
   
end_advanced_behavior
