include_once processors/quantum_agents/common/common_platform_script.txt
include_once processors/quantum_agents/aiai/state_machine_blue_checkpoint_1.txt
include_once processors/quantum_agents/aiai/state_machine_blue_checkpoint_2.txt
include_once processors/quantum_agents/aiai/state_machine_green_checkpoint_1.txt
include_once processors/quantum_agents/aiai/state_machine_green_checkpoint_2.txt
include_once processors/quantum_agents/aiai/state_machine_green_checkpoint_3.txt
include_once processors/quantum_agents/aiai/state_machine_red_checkpoint_1.txt
include_once processors/quantum_agents/aiai/state_machine_red_checkpoint_2.txt
include_once processors/quantum_agents/aiai/state_machine_red_checkpoint_3.txt
include_once processors/quantum_agents/aiai/state_machine_red_checkpoint_4.txt

platform_type  STRIKER  WSF_PLATFORM

   mover WSF_AIR_MOVER
      roll_rate_limit               1 rad/sec
      default_linear_acceleration   1.0 g
      default_radial_acceleration   6.5 g
      default_climb_rate            400 fps
      maximum_climb_rate            400 fps
      maximum_speed                 600.0 knots
      minimum_speed                 150.0 knots
      maximum_altitude              50000 ft
      minimum_altitude              50 ft
      maximum_linear_acceleration   9 g
      at_end_of_path extrapolate
      turn_rate_limit               4.0 deg/sec
   end_mover

   fuel WSF_FUEL
      consumption_rate     7 lb/s
      initial_quantity 14000 lb     #~33 minutes, at this consumption rate
      maximum_quantity 14000 lb
      bingo_quantity    1000 lb
   end_fuel  

   processor task_mgr WSF_STATE_MACHINE
      update_interval 5 sec
   
      script_variables
         string currentState = "checkpoints_blue";
      end_script_variables
   
      state checkpoints_blue
         next_state checkpoints_green         
            if (PLATFORM.GroundRangeTo(WsfSimulation.FindPlatform("Blue_checkpoint_2")) < 10000)
            {
               currentState = "checkpoints_green";
               return true;
            }
            return false;
         end_next_state
         
         next_state checkpoints_blue
            return true;
         end_next_state
         
         advanced_behavior_tree
            btt on
            sequence
               behavior_node blue_checkpoint_1
               behavior_node blue_checkpoint_2
            end_sequence
         end_advanced_behavior_tree
      end_state
      
      state checkpoints_green
         next_state checkpoints_red
            if (PLATFORM.GroundRangeTo(WsfSimulation.FindPlatform("Green_checkpoint_3")) < 10000)
            {
               currentState = "checkpoints_red";
               return true;
            }
            return false;
         end_next_state
      
         next_state checkpoints_green
            return true;
         end_next_state
         
         advanced_behavior_tree
            btt on
            sequence
               behavior_node green_checkpoint_1
               behavior_node green_checkpoint_2
               behavior_node green_checkpoint_3
            end_sequence
         end_advanced_behavior_tree
      end_state
      
      state checkpoints_red
         next_state checkpoints_red
            return true;
         end_next_state
         
         advanced_behavior_tree
            btt on
            sequence
               behavior_node red_checkpoint_1
               behavior_node red_checkpoint_2
               behavior_node red_checkpoint_3
               behavior_node red_checkpoint_4
            end_sequence
         end_advanced_behavior_tree
      end_state
      
      on_update

      
            currentState = STATE_MACHINE.Evaluate(currentState);
      end_on_update
   end_processor

end_platform_type
