platform_type PARACHUTE WSF_PLATFORM
   icon parachute
   mover WSF_GUIDED_MOVER
     update_interval .01 sec
     stage 1 # intial jump; parachute not open
        total_mass 0.01 lbs
        pre_separation_coast_time   10 sec
        aero INITIAL_PARACHUTE_AERO
     end_stage
     stage 2 # parachute open
        total_mass 20 lbs
        aero PARACHUTE_AERO
     end_stage
   end_mover

   processor guidance WSF_GUIDANCE_COMPUTER
      #show_status
      #show_evaluations
      #debug
      phase LAUNCH
         guidance_delay  10 sec
         #maximum_descent_rate   5 ft/s
         next_phase OPEN when phase_time > 10 sec
      end_phase
      phase OPEN
         guidance_delay  50000 sec
         #maximum_descent_rate   5 ft/s
      end_phase
   end_processor
end_platform_type
