# Test-sender ###################################################################
# This platform is used to test the iridium communications.
platform sender1 WSF_PLATFORM

  commander SELF
  side blue
  icon cmd_post
  position 20n 80e altitude 100 ft

  add comm sender1 WSF_RADIO_TRANSCEIVER

     on
     network_name sender1

   transmitter
      antenna_pattern                     UNIFORM_0DBI_PATTERN
      attenuation                         none
      frequency                           1621 mhz  # ref 1
      bandwidth                           10.5 mhz  # ref 1
      propagation_model                   none
      polarization                        right_circular  # ref 1
      power                               50 dbw
   end_transmitter

   receiver
      antenna_pattern                     UNIFORM_0DBI_PATTERN
      frequency                           1621 mhz  # ref 1
      bandwidth                           10.5 mhz  # ref 1
      noise_power                         -105.0 dbm
      detection_threshold                 3.0 dB
   end_receiver

  end_comm

   add sensor detect DETECT
      internal_link linker
   end_sensor

  add processor linker DYNAMIC_LINKING_PROC
  end_processor

   // test script
   execute at_interval_of 1 s
      if (TIME_NOW > 20.0)
      {
         WsfControlMessage msg = WsfControlMessage();
         msg.SetFunction("blah blah blah");
         WsfPlatform commander2 = WsfSimulation.FindPlatform("commander2");
         WsfComm comm = PLATFORM.Comm("sender1");
         if(comm.PathExistsTo(commander2.Comm("commander2").GetAddress()))
         {
            PLATFORM.Comm("sender1").SendMessage(msg, commander2.Comm("commander2").GetAddress());
         }
      }
   end_execute

end_platform
