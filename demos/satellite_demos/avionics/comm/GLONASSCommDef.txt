comm GLONASS_DOWNLINKL1_TX WSF_RADIO_XMTR

   azimuth_field_of_view                  -180.0 deg 180.0 deg
   elevation_field_of_view                 -90.0 deg   -70.0 deg  # ref 8


   transmitter
      antenna_pattern                     SATELLITE_NADIR_CONE
      attenuation                         none
      frequency                           1602 MHz  # ref 1, Each GLONASS satellite uses a unique frequency 1602 + n x 0.5625
      bandwidth                           6.5  MHz # ref 6
      wavelength                          18.7 cm   # ref 6
      propagation_model                   none
      power                               26.0 dBW   # ref 3
      pulse_width                         15.0 usec
      pulse_repetition_interval           150.0 usec
   end_transmitter

end_comm
