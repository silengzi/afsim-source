###############################################################################
# 4th_constellation_demo.txt
# This example demonstrates how to easily create a notional satellite
# constellation with only a basic understanding of orbital elements.
#
# Duplicate the GPS constellation
#
# MEO: SMA 26600 km
#    (2 Sidereal Revs/Day)
# 4 satellites in 6 orbits
# 55 degree inclination
# 60 degree separations in right ascension of ascending node          (x6 = 360 deg)
# 90 degree anomaly separations (x4)
# Note: One may have to play with the anomalies to get the phases correct among
#       the 4 GPS orbits.
###############################################################################

define_path_variable CASE constellation_demo
log_file output/$(CASE).log

file_path .
file_path ../base_types

include event_pipe.txt
include event_output.txt
include terrain.txt

# include the platform type definitions
include_once platforms/simple_gps_satellite.txt

# Optionally, uncomment the following for a "reality check":
# include_once scenarios/gps_satellites.txt

#The event and replay file are stated last just in
#case a previous file setup default files.

# define the platform type with the SMA we need
platform_type NOTIONAL_GPS WSF_PLATFORM
   side blue
   icon satellite
   mover WSF_SPACE_MOVER
      revs_per_day 2.0
      inclination 55 deg
   end_mover
end_platform_type

# First orbit, 4 anomalies (0, 90, 180, 240 deg) at raan = 0 degrees
platform gps-11 NOTIONAL_GPS
