radar_signature TARGET_RADAR_SIGNATURE
  constant 10 dbsm
end_radar_signature

platform_type TARGET_PLATFORM_TYPE WSF_PLATFORM
   radar_signature TARGET_RADAR_SIGNATURE
end_platform_type

antenna_pattern BISTATIC_TX_ANTENNA
  sine_pattern
    peak_gain              15 dB
    minimum_gain           -20.0 db
    azimuth_beamwidth      30 deg
    elevation_beamwidth    20 deg
end_antenna_pattern

antenna_pattern BISTATIC_RX_ANTENNA
  sine_pattern
    peak_gain              20 dB
    minimum_gain           -20.0 db
    azimuth_beamwidth      20 deg
    elevation_beamwidth    20 deg
end_antenna_pattern

sensor BISTATIC_SENSOR_RCVR WSF_RADAR_SENSOR
   receive_only

   antenna_height                 25.0 m

   frame_time                     10.0 sec

   receiver
      antenna_pattern             BISTATIC_RX_ANTENNA
      beam_tilt                   0 deg
      frequency                   400 mhz
      bandwidth                   2.0 mhz
      noise_power                 -175 dBw
   end_receiver

   detection_threshold 3 db
   #probability_of_false_alarm     1.0e-6
   #required_pd                    0.9
   #swerling_case                  1

   reports_range
   reports_bearing
   reports_elevation
   reports_iff
end_sensor

sensor BISTATIC_SENSOR_XMTR WSF_RADAR_SENSOR
   transmit_only

   antenna_height                 25.0 m

   frame_time                     10.0 sec

   transmitter
      antenna_pattern             BISTATIC_TX_ANTENNA
      beam_tilt                   25 deg
      power                       100.0 w
      frequency                   400 mhz
      bandwidth                   2.0 mhz
   end_transmitter

   reports_range
   reports_bearing
   reports_elevation
   reports_iff
end_sensor

platform bistatic_rcvr WSF_PLATFORM
   position 0n 0e heading 90 deg
   add sensor bistatic_rcvr BISTATIC_SENSOR_RCVR
      on
      #debug
   end_sensor
