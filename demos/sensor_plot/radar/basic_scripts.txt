# pd
script double pd(WsfPlatform aPlatform, WsfSensor aSensor, WsfPlatform aTarget, WsfSensorInteraction aResult, double aCurrentValue)
   if (aResult.Succeeded() &&
       (aResult.Pd() > aResult.RequiredPd()))
   {
      return MATH.Max(aCurrentValue, aResult.Pd());
   }
   return aCurrentValue;
end_script

# detection_threhold
script double detection_threshold(WsfPlatform aPlatform, WsfSensor aSensor, WsfPlatform aTarget, WsfSensorInteraction aResult, double aCurrentValue)
   if (aResult.DetectionThreshold() > 0.0)
   {
      double value = aResult.DetectionThreshold();
      if (Math.Fabs(aCurrentValue -300.0) > 0.001)
      {
         return MATH.Min(aCurrentValue, value);
      }
      return value;
   }
   return aCurrentValue;
end_script

# detection_threhold
script double required_rcs(WsfPlatform aPlatform, WsfSensor aSensor, WsfPlatform aTarget, WsfSensorInteraction aResult, double aCurrentValue)
   if ((aResult.SignalToNoise() > -300.0) &&
       (aResult.RadarSignature() > 0.0))
    {
       double value = Math.LinearToDB(aResult.RadarSignature()) + (mRequiredRCS_DetectionReference - aResult.SignalToNoise());
       return MATH.Min(aCurrentValue, value);
    }
   return aCurrentValue;
end_script

       (aResult.RcvdPower() > -300.0) &&

   return aCurrentValue;
end_script

# jammer_power
script double jammer_power(WsfPlatform aPlatform, WsfSensor aSensor, WsfPlatform aTarget, WsfSensorInteraction aResult, double aCurrentValue)
   if ((aResult.NoiseJammerPower() > -300.0) ||
       (aResult.PulseJammerPower() > -300.0))
   {
      double jamPower = 0.0;
      if (aResult.NoiseJammerPower() > -300.0)
      {
         jamPower = jamPower + Math.DB_ToLinear(aResult.NoiseJammerPower());
      }
      if (aResult.PulseJammerPower() > -300.0)
      {
         jamPower = jamPower + Math.DB_ToLinear(aResult.NoiseJammerPower());
      }
      return Math.Max(aCurrentValue, Math.LinearToDB(jamPower));
   }
   return aCurrentValue;
end_script

# jammer_to_signal
script double jammer_to_signal(WsfPlatform aPlatform, WsfSensor aSensor, WsfPlatform aTarget, WsfSensorInteraction aResult, double aCurrentValue)
   if ((aResult.RcvdPower() > -300.0) &&
       ((aResult.NoiseJammerPower() > -300.0) ||
        (aResult.PulseJammerPower() > -300.0)))
   {
      double jamPower = 0.0;
      if (aResult.NoiseJammerPower() > -300.0)
      {
         jamPower = jamPower + Math.DB_ToLinear(aResult.NoiseJammerPower());
      }
      if (aResult.PulseJammerPower() > -300.0)
      {
         jamPower = jamPower + Math.DB_ToLinear(aResult.NoiseJammerPower());
      }
      return Math.Max(aCurrentValue, Math.LinearToDB(jamPower) - aResult.RcvdPower());
   }
   return aCurrentValue;
end_script
