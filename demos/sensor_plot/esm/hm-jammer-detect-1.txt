include signatures/ucav_radar_sig.txt
include signatures/ucav_infrared_sig.txt
include signatures/ucav_optical_sig.txt

ignore_block route
ignore_block mover

antenna_pattern  constAntenna
  constant 3 db
end_antenna_pattern

sensor PASSIVE_SNSR WSF_ESM_SENSOR
  frame_time    0.5 sec
  azimuth_field_of_view     -180 deg 180 deg
  elevation_field_of_view   -90 deg 90 deg
  frequency_band   1.0 ghz  100.0 ghz

  receiver
    antenna_pattern constAntenna
    noise_power   -120 dbm
  end_receiver
end_sensor

platform_type TARGET_PLATFORM_TYPE WSF_PLATFORM
   radar_signature    UCAV_RADAR_SIG
   optical_signature  UCAV_OPTICAL_SIG
   infrared_signature UCAV_INFRARED_SIG

   sensor passive PASSIVE_SNSR
      on
      reports_location
      reports_signal_to_noise
   end_sensor

end_platform_type

antenna_pattern JAMMER_ANTENNA
   sine_pattern
      peak_gain             10.0 db
      minimum_gain          -20.0 db
      azimuth_beamwidth     20.0 deg
      elevation_beamwidth   20.0 deg
   end_sine_pattern
end_antenna_pattern

weapon JAMMER WSF_RF_JAMMER
   transmitter
      antenna_pattern            JAMMER_ANTENNA
      frequency                  3 ghz
      bandwidth                  10 mhz
      power                      100.0 w
      internal_loss              0.0 db
   end_transmitter
end_weapon

platform jammer-1 WSF_PLATFORM

   position 0.0n 0.0e
   altitude 30000 ft
   heading  0 deg

   add weapon jammer JAMMER
      on

      yaw   0.0 deg
      pitch 0.0 deg
      roll  0.0 deg
   end_weapon

end_platform
