mover SR_SAM_MOVER WSF_GUIDED_MOVER
   update_interval                  0.01 sec
   integration_timestep             0.01 sec

   // ALL MADE UP to get about 10nm
   // Total Mass   = 165 kg
   // Warhead Mass = 15 kg
   // Mass Fraction = .9 == 140 x .85 = 126 kg of fuel
   // So 165-15
   // Thrust Duration = 2 sec
   // thrust  46000 nt
   stage 1
      aero                          SR_SAM_AERO
      total_mass                    165 kg
      fuel_mass                     130 kg
      thrust                        46000 nt
      thrust_duration               2 sec
   end_stage

end_mover

processor SR_SAM_AUTOPILOT WSF_GUIDANCE_COMPUTER
   #show_status
   guide_to_truth true
   #show_evaluations

   phase LAUNCH_AND_PITCH_OVER
      on_entry
         WsfTrack targetTrack = PLATFORM.CurrentTargetTrack();
         double targetSlantRange = PLATFORM.SlantRangeTo(targetTrack);
	     if (targetSlantRange < 7408)
	     {
	        SelectPhase("SHORT_RANGE");
	     }
      end_on_entry
      maximum_commanded_g 35 g
      commanded_flight_path_angle   45 deg
      next_phase MIDCOURSE when on_commanded_flight_path_angle
   end_phase

   phase SHORT_RANGE
      maximum_commanded_g 40 g
      commanded_flight_path_angle   20 deg
      next_phase MIDCOURSE when on_commanded_flight_path_angle
   end_phase

   phase MIDCOURSE
      #on_entry
      #   //SetDebugEnabled(true);
      #end_on_entry
	  maximum_commanded_g 35 g
      proportional_navigation_gain  4.0
	  g_bias 1.1
      next_phase TERMINAL when target_ground_range < 4 km
   end_phase

   phase TERMINAL
      #on_entry
      #   #SetDebugEnabled(true);
      #end_on_entry
	  maximum_commanded_g 40 g
      proportional_navigation_gain  4.0
   end_phase
end_processor
