script WsfBMInterceptCalcResult SAM_LaunchComputer(WsfTrack aTrack,
                                                   WsfPlatform aLauncher,
                                                   string aEnvelopeName,
                                                   double aLaunchDelay,
                                                   double aPK)
   WsfBMInterceptCalcResult result = WsfBMInterceptCalcResult();
   result.SetCantIntercept();

   double tgtAlt = aLauncher.RelativeAltitudeOf(aTrack);

   double maxRng = 25.0 * 1852.0; # 25 nautical miles
   double maxTOF = 27.0;
   double avgSpd = 2643.0 * 1852.0 / 3600.0; # kts -> m/sec

   WsfWaypoint intercept = WsfWaypoint();
   double tti = WsfBMUtils.InterceptLocation2DNoProject(aTrack, aLauncher, intercept, avgSpd);

   if ((tti > 0.0) && (tti < maxTOF))
   {
      result.SetCanIntercept(true);
      result.SetTTI(tti);

      double lat = intercept.Latitude();
      double lon = intercept.Longitude();
      WsfGeoPoint point;
      point = point.Construct(lat, lon, tgtAlt);
      result.SetPredPoint(point);

      result.SetWeaponMaxRange(maxRng);
      result.SetEstimatedPK(aPK);
   }

   return result;
end_script
