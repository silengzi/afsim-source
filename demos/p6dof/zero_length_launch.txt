# In this scenario, an FA-LGT aircraft is launched using a Zero Length Launch
# (ZELL) technique. ZELL was developed during the 1950's to allow aircraft to
# takeoff without the use of a runway. ZELL involves attaching a large solid-
# propellant rocket to the bottom of a fighter. The rocket is ignited and
# provides thrust to accelerate the fighter to flying speed before the rocket
# is jettisoned.
#
# This demo shows how a P6DOF sub-object can command itself to be launched
# from a parent object. The zell launch platform has a sub-object fighter of
# type FA-LGT-ZLL. A sequencer on the fighter is activated 5 seconds after
# the FA-LGT-ZLL platform is instantiated. This sequencer ignites the attached
# rocket engine and separates the fighter from the launcher. Three seconds
# later, after the rocket burns out, the rocket is jettisoned from the fighter.
#
# ******************************************************************************

define_path_variable CASE zero_length_launch

include_once std_setup.txt

# Set start time, GMT (The start date was set in std_setup.txt above)
start_time 23:00:00.000

end_time 90 sec

include_once p6dof_types/environment/p6dof_environment.txt
include_once platforms/zell-launcher.txt

platform zell_launch_platform ZELL-LAUNCHER
   update_interval            0.01 sec
   side                       blue

   p6dof_position             21.301 -158.074

   p6dof_alt                  10.0 ft
   p6dof_set_velocity_ned_fps 0.0 0.0 0.0
   p6dof_ned_heading          200.0 deg
   p6dof_ned_pitch            0.0 deg
   p6dof_ned_roll             0.0 deg
end_platform

route SimpleRoute
   label start
   position   21.330n  158.130w  altitude 5000.0 ft speed 500.0 kts
   position   21.420n  158.130w  altitude 5000.0 ft speed 500.0 kts
   position   21.420n  158.040w  altitude 5000.0 ft speed 500.0 kts
   position   21.320n  158.040w  altitude 5000.0 ft speed 500.0 kts
   goto start
end_route

script void PlatformInitialized(WsfPlatform aPlatform)

   writeln("PlatformInitialized at ", TIME_NOW, "sec ... ", aPlatform.Name());

   if(aPlatform.Name() == "zell_launch_platform_Fighter")
   {
      WsfMover mover = aPlatform.Mover();
      if(mover.IsA_TypeOf("WSF_P6DOF_MOVER"))
      {
         WsfP6DOF_Mover p6dofMover = (WsfP6DOF_Mover)mover;
         p6dofMover.EnableControls(false);
         p6dofMover.EnableAutopilot(false);
      }

      writeln("  Goto 1,000 ft, 200 kts, and head north...  ");

     WsfSimulation.ExecuteAtTime(TIME_NOW + 2.0, "PullBackOnStick", {aPlatform});
     WsfSimulation.ExecuteAtTime(TIME_NOW + 5.0, "HeadEast", {aPlatform});
   }
end_script

script void PullBackOnStick(WsfPlatform aPlatform)
   writeln("PullBackOnStick at ", TIME_NOW, " sec ...");

   WsfMover mover = aPlatform.Mover();
   if(mover.IsA_TypeOf("WSF_P6DOF_MOVER"))
   {
      WsfP6DOF_Mover p6dofMover = (WsfP6DOF_Mover)mover;

      writeln("Enable controls but keep AP disabled...");
      p6dofMover.EnableControls(true);
      p6dofMover.EnableAutopilot(false);
      p6dofMover.TakeDirectControlInput();
      p6dofMover.SetDirectControlInputs(0.0,1.0,0.0,2.0); // Right,Back,Rudder,Throttle
   }
end_script

script void HeadEast(WsfPlatform aPlatform)
   writeln("HeadEast at ", TIME_NOW, " sec ...");

   WsfMover mover = aPlatform.Mover();
   if(mover.IsA_TypeOf("WSF_P6DOF_MOVER"))
   {
      WsfP6DOF_Mover p6dofMover = (WsfP6DOF_Mover)mover;

      writeln("Enable the AP...");
      p6dofMover.ReleaseDirectControlInput();
      p6dofMover.EnableControls(true);
      p6dofMover.EnableAutopilot(true);
   }

   aPlatform.GoToAltitude(2500*MATH.M_PER_FT());
   aPlatform.GoToSpeed(450*MATH.MPS_PER_NMPH());
   aPlatform.TurnToHeading(10);

end_script

script void FollowNewRoute(WsfPlatform aPlatform)
   writeln("FollowNewRoute at ", TIME_NOW, " sec ...");

   WsfMover mover = aPlatform.Mover();
   if(mover.IsA_TypeOf("WSF_P6DOF_MOVER"))
   {
      WsfP6DOF_Mover p6dofMover = (WsfP6DOF_Mover)mover;

      writeln("Enable controls but keep AP disabled...");
      p6dofMover.EnableControls(true);
      p6dofMover.EnableAutopilot(true);
      p6dofMover.TakeDirectControlInput();
      p6dofMover.SetDirectControlInputs(0.0,0.0,0.0,2.0); // Right,Back,Rudder,Throttle
   }

end_script

observer
   enable PLATFORM_INITIALIZED PlatformInitialized
end_observer
