# Create a script processor to add to any weapon platform
# This script example writes out body information and relative geometries
# between the a platform (typically a weapon) and the target
# track

processor WRITE_INFO WSF_SCRIPT_PROCESSOR
  update_interval 1 sec
  on_initialize # Writes out the header one time for the data
    writeln("Name,Mtime,Vel(mps),Mach,Alt,WCSVelocity(x,y,z),RelAz,RelEl");
  end_on_initialize
  on_update
    Vec3 vectorVelocityWCS = PLATFORM.VelocityWCS(); # get my xyz velocities
    WsfTrack track = PLATFORM.CurrentTargetTrack();  # get my current track info
    WsfPlatform target = track.Target();             # get my target info from the track
    // Do a validity check on the track and target so we don't spew NULL warning messages
    // out because the target platform no longer exists (because we destoyed it).
    if(track.IsValid() && target.IsValid())
    {
       WsfGeoPoint trackpoint = track.ReportedLocation(); # get the geolocation of the track
       double RelativeAzimuth = PLATFORM.RelativeAzimuthOf(trackpoint);
       double RelativeElevation = PLATFORM.RelativeElevationOf(trackpoint);
       double weaponAz = target.RelativeAzimuthOf(PLATFORM.Location());
       double weaponEl = target.RelativeElevationOf(PLATFORM.Location());
       writeln(PLATFORM.Name(),",",
            PLATFORM.TimeSinceCreation(),",",
            PLATFORM.Speed(),",",
            PLATFORM.MachNumber(),",",
            PLATFORM.Altitude(),",",
            vectorVelocityWCS,",",
            RelativeAzimuth,",",
            RelativeElevation,",",
            weaponAz,",",
            weaponEl );
    }
  end_on_update
end_processor

processor BLUE_GLIDE_BOMB_1_LAUNCH_COMPUTER WSF_TASK_PROCESSOR
#   script_debug_writes on
#   show_state_transitions
#   show_task_messages

   script_variables
      string WEAPON_NAME               = "blue_glide_bomb_1";
      int SALVO_SIZE                   = 1;
      # the following variable is used internally and should not be modified by the
      # user
      string mShootTaskStr      = "Shoot";
   end_script_variables

    script bool InInterceptEnvelopeOf(WsfWeapon aWeapon)
       bool canIntercept = false;
       double maxRng = 74080; #40 nm
       double minRng = 300;
       WsfTrackId id = TRACK.TrackId();
       double targetrange = PLATFORM.GroundRangeTo(TRACK);
       if ((targetrange > minRng) && (targetrange < maxRng))
       {
         canIntercept = true;
         writeln_d("Target is in range to intercept");
       }
       return canIntercept;
   end_script

   script bool HaveWeapons()
      WsfWeapon weapon = PLATFORM.Weapon(WEAPON_NAME);
      bool haveWeapons = false;
      if (weapon.QuantityRemaining() >= SALVO_SIZE)
      {
          haveWeapons = true;
      }
      return haveWeapons;
   end_script

   script bool LaunchWeapon()
      WsfWeapon   weapon;
      weapon = PLATFORM.Weapon(WEAPON_NAME);
      bool canInterceptNow = false;
      if (HaveWeapons())
      {
         canInterceptNow = InInterceptEnvelopeOf(weapon);
      }
      bool launched = false;
      if (canInterceptNow)
      {
         launched = FireAt(TRACK, mShootTaskStr, weapon, SALVO_SIZE);
         if (launched)
         {
            writeln("*** T=", TIME_NOW, " ", PLATFORM.Name(), " ",
                      TRACK.TargetName(), " R=", PLATFORM.SlantRangeTo(TRACK),
                      " FIRE!!!!");
         }
      }
      return launched;
   end_script


# State Machine
# Very simple state machine
  evaluation_interval GOHOME 1.0 sec
  state GOHOME  # Goes back to Loiter if out of weapons
      next_state ENGAGE
         bool weaponavail = true;
         if (! HaveWeapons())
         {
             weaponavail = false;
             PLATFORM.ReturnToRoute();
             writeln_d("Turning off launch computer");
             PROCESSOR.TurnOff();
         }
         return weaponavail;
      end_next_state
   end_state

   evaluation_interval ENGAGE 1.0 sec
   state ENGAGE  # Engages and fires the weapon if it can
      next_state GOHOME
         bool launched = false;
         WsfWeapon weapon = PLATFORM.Weapon(WEAPON_NAME);
         if ((InInterceptEnvelopeOf(weapon)) && weapon.TimeSinceLastFired() > weapon.FiringInterval())
         {
             writeln_d("Trying to launch weapon " );
             launched = LaunchWeapon();
         }
         return launched;
      end_next_state
   end_state
end_processor

# Define the platform type for the notional fighter/bomber.
platform_type FIGHTER_BOMBER WSF_PLATFORM
   category no_awacs_esm
   icon FIGHTER
   mover WSF_AIR_MOVER
     maximum_radial_acceleration 1.12 g
     maximum_linear_acceleration 2 g
   end_mover

   comm rcvr-1 WSF_RADIO_RCVR
      frequency 1200 mhz
      processor track-proc
      processor attack-proc
   end_comm

   comm xmtr-1 WSF_RADIO_XMTR
      frequency 1200 mhz
   end_comm

   processor track-proc WSF_TRACK_PROCESSOR
   end_processor

   processor attack-proc WSF_SCRIPT_PROCESSOR

      on_message
         type WSF_TRACK_MESSAGE
            script
               WsfTrackMessage trackMsg = (WsfTrackMessage)MESSAGE;
               WsfTrack        track    = trackMsg.Track();
               WsfTrackId      trackId  = track.TrackId();
               double     targetLat     = track.Latitude();
               double     targetLon     = track.Longitude();
               if (TIME_NOW < 50.0)
               {
                  writeln("Received track message from ", trackId.OwningPlatform());
               }

               // If we aren't attacking then start the 'attack'
               WsfProcessor launchComputer = PLATFORM.Processor("blue_glide_bomb_1-launch-computer");
               if (! launchComputer.IsTurnedOn())
               {
                  writeln_d("Starting attack at T=", TIME_NOW);
                  launchComputer.TurnOn();
                  PLATFORM.GoToLocation(targetLat, targetLon);
               }
            end_script

         type default
            script
               writeln_d("Received message of type ", MESSAGE.Type());
            end_script
      end_on_message
   end_processor

   weapon blue_glide_bomb_1 BLUE_GLIDE_BOMB_1
   end_weapon

   processor blue_glide_bomb_1-launch-computer BLUE_GLIDE_BOMB_1_LAUNCH_COMPUTER
     off
     #on
     update_interval 1 sec
   end_processor
end_platform_type
