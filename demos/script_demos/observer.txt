#### Example Observer to play sounds when certain events occur
#### Jan 2015
#### CHANGE THE PATH TO WHERE THE SOUND FILES ARE. WMPLAYER doesn't use relative pathing...

script void PlaySoundBoom()
   ### The following is for Out of the Box Windows 7...
   ### Windows XP used "sndrec32.exe /embedding /play /close <file_name>" and played in the background without a gui...
   //string str = write_str("wmplayer ..\\script_demos\\explosion.wav");
   //writeln(str);  # just to show what we are doing to make the sound
   ### Windows Media Player is NOT my first choice to play audio files...
   System.Command("start wmplayer ..\\script_demos\\explosion.wav");
end_script

script void PlaySoundBombDrop()
   ### The following is for Out of the Box Windows 7...
   //string str = write_str("wmplayer ..\\script_demos\\dropbomb.wav");
   //writeln(str);  # just to show what we are doing to make the sound
   System.Command("start wmplayer ..\\script_demos\\dropbomb.wav");
end_script


script void WeaponTerminated(WsfWeaponEngagement aWeaponEngagement)
   PlaySoundBoom();
end_script

script void PlatformKilled(WsfPlatform aPlatform)
   PlaySoundBoom();
end_script

script void WeaponHit(WsfWeaponEngagement aWeaponEngagement, WsfPlatform aTargetPlatform)
   PlaySoundBoom();
end_script

script void WeaponFired(WsfWeaponEngagement aWeaponEngagement, WsfTrack aTargetTrack)
   PlaySoundBombDrop();
end_script

observer
   enable WEAPON_TERMINATED
   enable PLATFORM_BROKEN
   enable WEAPON_HIT
   enable WEAPON_FIRED
end_observer
