# we must first pull in some DTED data:

   raw_dted 2 dted 39 -107 40 -106
end_terrain

   update_interval    5 sec
end_laser_designations

end_platform

end_platform

event_output
   file rf_auto_test.evt
   enable WEAPON_TERMINATED
   enable WEAPON_FIRED
   enable PLATFORM_BROKEN
   enable SENSOR_MODE_ACTIVATED
   enable SENSOR_TRACK_INITIATED
end_event_output

event_pipe 
   file rf_auto_test.aer
   use_preset full
end_event_pipe

end_time 15 min
