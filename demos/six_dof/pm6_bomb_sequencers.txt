# This scenario demonstrates five examples of bombing using SixDOF movers.
#
# The first example uses a heavy bomber (Rolling Thunder) at very low altitude
# that drops a series of high-drag bombs to destroy a runway.
#
# The second example uses a heavy bomber (Arc Light) at high altitude that
# drops a series of low-drag bombs to destroy another runway. This aircraft
# is in the contrail band and will produce contrails in the Warlock and
# Mystic tools.
#
# The third example uses a flying-wing bomber (Ghost) at medium altitude that
# drops a series of guided-bombs (representing GPS-guided bombs) that use
# pre-briefed tracks to attack point targets at the air base.
#
# The fourth example uses a tactical fighter (Viper 1) to drop low-drag bombs
# using a dive bombing approach. This shows the use of a single, nested
# sequencer to drop multiple bombs with a single call.
#
# The fifth (and last) example also uses a tactical fighter (Viper 2) to drop
# low-drag bombs with a dive bombing approach. In this case, however, it uses
# individual sequencers to drop bombs individually.
#
# ******************************************************************************

platform arc_light B-HVY-BOMBS-LD
   update_interval 0.01 sec
   side blue

   route
      position 21:14:10.0n 158:10:29.0w  altitude 36000.0 ft  speed 530.0 kts
      position 21:18:10.0n 158:04:29.0w  altitude 36000.0 ft  speed 530.0 kts
   end_route

   edit mover
      wash_in_conditions true
   end_mover   

   script_variables
      bool bombTriggerPulled = false;
      bool bombsRemain       = true;
      string activeRack      = "iron_bomb_aft_rack";
   end_script_variables
   
   script void ReleaseNextBomb()

      // First, verify bombs are available
      WsfWeapon wpn = PLATFORM.Weapon(activeRack);
      if(wpn.IsValid())
      {
         if(wpn.QuantityRemaining() < 0.5)
         {
            // Switch to the next rack
            if(activeRack == "iron_bomb_aft_rack")
            {
               activeRack = "iron_bomb_mid_rack";
            }
            else if(activeRack == "iron_bomb_mid_rack")
            {
               activeRack = "iron_bomb_forward_rack";
            }
            else
            {
               // No racks
               bombsRemain = false;
            }
         }
      }
      
      if(bombTriggerPulled && bombsRemain)
      {
         WsfWeapon wpn = PLATFORM.Weapon(activeRack);
         if(wpn.IsValid())
         {
            if(wpn.QuantityRemaining() > 0.0)
            {
               wpn.Fire();
            }
         }
      }

      if(bombsRemain) { PLATFORM.ExecuteAtTime(TIME_NOW + 0.3, "ReleaseNextBomb", {}); }
      
   end_script
   
   script void ReleaseBombs()
      bombTriggerPulled = true;
      ReleaseNextBomb();
   end_script
   
   # Drop the bombs
   execute at_time 5.0 sec absolute   
      writeln("Bomber 'arc_light' drops its bombs...");
      ReleaseBombs();
   end_execute

   execute at_time 16.0 sec absolute
      writeln("Bomber 'arc_light' making a right turn towards the east...");
      PLATFORM.TurnToHeading(90.0);
   end_execute

   execute at_time 35.0 sec absolute
      writeln("Bomber 'arc_light' continuing turn towards the southwest...");
      PLATFORM.TurnToHeading(230.0);    
   end_execute

end_platform
