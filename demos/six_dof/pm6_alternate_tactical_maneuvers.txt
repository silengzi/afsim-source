script Array<double> DetermineRollAndPull(WsfPlatform aAttackingPlatform, WsfPlatform aTargetPlatform)
   WsfGeoPoint tgtPt = aTargetPlatform.Location();
   double az_deg = aAttackingPlatform.RelativeAzimuthOf(tgtPt);
   double el_deg = aAttackingPlatform.RelativeElevationOf(tgtPt);
   double CosEl = Math.Cos(el_deg*Math.RAD_PER_DEG());
   double SinEl = Math.Sin(el_deg*Math.RAD_PER_DEG());
   double x  = Math.Cos(az_deg*Math.RAD_PER_DEG());
   double y  = Math.Sin(az_deg*Math.RAD_PER_DEG());
   double xy = Math.Sqrt( x*x + y*y );
   double z  = 0.0;
   if( Math.Fabs(CosEl) > 0.0000001 )
   {
      z  = (xy/CosEl) * SinEl;
   }
   double yz = Math.Sqrt( y*y + z*z );
   double ang_deg = Math.DEG_PER_RAD() * Math.ATan2(yz,x);

   double tgtRoll_deg    = 90.0 - Math.ATan2(z,y);

   Array<double> retval = Array<double>();
   retval.PushBack(tgtRoll_deg);
   retval.PushBack(ang_deg);
   return retval;
end_script

script double RollRateFromTargetRoll(double aTargetRoll)
   double rollRate_dps = 0.01 * aTargetRoll * Math.Fabs(aTargetRoll);
   if( rollRate_dps > 360.0 )
   {
      rollRate_dps = 360.0;
   }
   if( rollRate_dps < -360.0 )
   {
      rollRate_dps = -360.0;
   }
   return rollRate_dps;
end_script

script double GLoadFromTotalAngle(double aTotalAngle)
   double gLoad = 1.0;
   if( aTotalAngle > 90.0 )
   {
      aTotalAngle = 90.0;
   }

   if( aTotalAngle < 5.0 )
   {
      gLoad = 0.25;
   }
   else
   {
      if( aTotalAngle > 45.0 )
      {
         gLoad = 6.5;
      }
      else if( aTotalAngle > 30.0 )
      {
         gLoad = 4.5;
      }
      else if( aTotalAngle > 20.0 )
      {
         gLoad = 3.0;
      }
      else if( aTotalAngle > 10.0 )
      {
         gLoad = 2.0;
      }
      else // > 5 deg
      {
         gLoad = 1.0;
      }
   }
   return gLoad;
end_script

# This is the executive function for the WsfScriptedManeuver that the blue
# aircraft is executing. The current time and the mover to which the
# maneuver has been assigned is supplied to the script. The only requirement
# of the script is that it return the next time at which the maneuver's
# executive function should be called.
script double ExecuteBluePursuit(double aSimTime, WsfSixDOF_Mover aMover)
   WsfPlatform tgtPlatform = WsfSimulation.FindPlatform("ManeuverFighter");
   WsfPlatform platform = aMover.Platform();

   Array<double> angles = DetermineRollAndPull(platform, tgtPlatform);

   double rollRate = RollRateFromTargetRoll(angles[0]);
   SetRollRate(platform, rollRate);

   double gLoad = GLoadFromTotalAngle(angles[1]);
   SetPitchGLoad(platform, gLoad);

   return aSimTime + 0.1;
end_script

   end_route

   execute at_time 0.1 sec absolute
      WsfSixDOF_Maneuver mvr = WsfSixDOF_ScriptedManeuver.Construct("ExecuteBluePursuit");
      WsfSixDOF_Mover mov = (WsfSixDOF_Mover)PLATFORM.Mover();
      mov.ExecuteManeuver(mvr);
   end_execute

end_platform

   script void SetInitialRanges(WsfSixDOF_Mover aMover)
      aMover.SetBankAngleMax(180.0);
      aMover.SetPitchGLoadMax(8.0);
      aMover.SetPitchGLoadMin(0.0);

   # This method and those that follow take the place of the scripts that
   # execute at the equivalent times in the original TacticalManeuvers
   # demo. Some of these return a single maneuver, and some a sequence
   # of maneuvers. All of these maneuvers are 'elemental' maneuvers: they
   # set a single channel on the autopilot as their entire operation.
   # More complex behaviors are pieced together out of elemental maneuvers.
   script WsfSixDOF_Maneuver FiveSeconds()
      WsfSixDOF_SpeedKCAS_Maneuver mvr = WsfSixDOF_SpeedKCAS_Maneuver.Construct(6000.0);
      mvr.SetEntryConstraint(WsfSixDOF_ManeuverConstraint.AT_TIME(5.0));
      return mvr;
   end_script

      retval.Append(WsfSixDOF_PitchDeltaManeuver.Construct(88.0));
      retval.Append(WsfSixDOF_RollDeltaManeuver.Construct(0.0));

   # This script will piece together the entire sequence of maneuvers
   # that the red aircraft is executing. Many of the returned values
   # from the methods above are sequences, showing that sequences of
   # sequences are supported.
   script WsfSixDOF_Maneuver CreateManeuverSequence()

      retval.Append(FiveSeconds());
      retval.Append(TwentySeconds());
      retval.Append(ThirtySeconds());
      retval.Append(FortySeconds());
      retval.Append(SeventySeconds());
      retval.Append(EightyFiveSeconds());
      retval.Append(NinetySeconds());

      retval.Append(ThreeHundredSeconds());
      retval.Append(ThreeHundredThirtySeconds());
      retval.Append(ThreeHundredFiftyFiveSeconds());
      retval.Append(ThreeHundredSixtySeconds());

      return retval;
   end_script
