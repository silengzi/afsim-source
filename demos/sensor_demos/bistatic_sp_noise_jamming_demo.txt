script void JammingAttempt(WsfPlatform       aXmtr,
                           WsfPlatform       aRcvr,
                           WsfEM_Interaction aResult)
   if (aResult.IsValid())
   {
      writeln("");
      writeln("------------------Jammer-----------------------");
      writeln("* RcvrToXmtrRange     = ", aResult.RcvrToXmtrRange()/1000., "\tkm\t*");
      writeln("* XmtrToRcvrRange     = ", aResult.XmtrToRcvrRange()/1000., "\tkm\t*");
      if (aResult.NoiseJammerPower() != 0.0)
      {
         writeln("* NoiseJammerPower    = ", aResult.NoiseJammerPower(), "\tdBW\t*");
      }
      if (aResult.PulseJammerPower() != 0.0)
      {
         writeln("* PulseJammerPower    = ", aResult.PulseJammerPower(), "\tdBW\t*");
      }
      if (aResult.CoherentJammerPower() != 0.0)
      {
        writeln("* CoherentJammerPower = ", aResult.CoherentJammerPower(), "\tdBW\t*");
        writeln("* RcvrNoisePower      = ", aResult.RcvrNoise(), "\tdBw\t*");
        writeln("* JNR                 = ", aResult.CoherentJammerPower() - aResult.RcvrNoise(), "\tdB\t*");
      }
      writeln("-----------------------------------------------");
      writeln("");
   }
end_script

script void SensorDetectionAttempt(WsfPlatform          aPlatform,
                                   WsfSensor            aSensor,
                                   WsfPlatform          aTarget,
                                   WsfSensorInteraction aResult)
   if (aResult.IsValid())
   {
      writeln("");
      writeln("-----------------------------------------------");
      writeln("* RcvrToTgtRange  = ", aResult.RcvrToTargetRange()/1000., "\tkm\t*");
      writeln("* XmtrToTgtRange  = ", aResult.XmtrToTargetRange()/1000., "\tkm\t*");
      writeln("* RcvdPower       = ", aResult.RcvdPower(), "\tdBW\t*");
      if (aResult.NoiseJammerPower() != 0.0)
      {
         writeln("* NoiseJammerPower = ", aResult.NoiseJammerPower(), "\tdBW\t*");
         writeln("* InterferenceNoise = ", MATH.LinearToDB(MATH.DB_ToLinear(aResult.RcvrNoise()) + MATH.DB_ToLinear(aResult.NoiseJammerPower())), "\tdBW\t*");
      }
      if (aResult.PulseJammerPower() != 0.0)
      {
         writeln("* PulseJammerPower = ", aResult.PulseJammerPower(), "\tdBW\t*");
      }
      if (aResult.CoherentJammerPower() != 0.0)
      {
        writeln("* CoherentJammerPower = ", aResult.CoherentJammerPower(), "\tdBW\t*");
        writeln("* RcvrNoisePower   = ", aResult.RcvrNoise(), "\tdBw\t*");
        writeln("* JNR              = ", aResult.CoherentJammerPower() - aResult.RcvrNoise(), "\tdB\t*");
      }
      writeln("* NoisePower      = ", aResult.RcvrNoise(), "\tdBw\t*");
      writeln("* SNR             = ", aResult.SignalToInterference(), "\tdB\t*");
      writeln("* BoltzmannConst  = ", MATH.BOLTZMANN_CONSTANT(), "\tJ/K\t*");
      writeln("* C(SpeedOfLight) = ", MATH.LIGHT_SPEED(), "\tm/s\t*");
      writeln("-----------------------------------------------");
      writeln("");
   }
end_script
