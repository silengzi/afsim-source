#
# This is a demo for a scanning EOIR sensor
#
# Note: this is accomplished by incrementally modifing the yaw
#       withing the "execute at_interval_of" script method (yaw_script)
#
#       While the "azimuth_slew_limits" need not match the limits
#       in the yaw_script, the values should be considered as they may
#       impose additional restrictions of movement.
#
# Caution: The one 1 s interval for yaw_script can be overtaxing in a
#          full scale scenario and increased as necessary.
#

platform test_scan_cam WSF_PLATFORM
   position 0n 0w heading 180 deg
   add sensor cam WSF_EOIR_SENSOR
      on
      azimuth_slew_limits -90 degrees 90 degrees
      elevation_slew_limits -20 degrees -5 degrees
      slew_mode both
      reports_location
      frame_time 1 s
      angular_resolution .1 degrees
      azimuth_field_of_view -10 deg 10 deg
      update_interval 1 s
      location 0 0 -100 m
      internal_link vid_proc
   end_sensor
   add processor vid_proc WSF_IMAGE_PROCESSOR
      internal_link proc
   end_processor
   add processor proc WSF_SCRIPT_PROCESSOR
      on_message
         type WSF_TRACK_MESSAGE
            script
               writeln(TIME_NOW, " detected Sensor Yaw: ", PLATFORM.Sensor("cam").Yaw());
            end_script
      end_on_message
   end_processor
   script bool IsBetween(double value, double min, double max)
      return (value >= min) && (value <= max);
   end_script

   execute at_interval_of 1 s
      static int sign = -1;
      double yaw = PLATFORM.Sensor("cam").Yaw();
      writeln(TIME_NOW, " Yaw ", yaw);
      if (!IsBetween(yaw, -90, 90))
      {
         sign = -sign;
         writeln("Sign ", sign);
      }
      PLATFORM.Sensor("cam").SetYaw(yaw + sign * MATH.PI()/32);
   end_execute
end_platform

platform target WSF_PLATFORM
   position 0n 0:3:10e
end_platform

end_time 60 m
