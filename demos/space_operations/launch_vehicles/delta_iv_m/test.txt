include delta_iv_m.txt

platform_type DELTA_IV_M_MOD DELTA_IV_M
   edit mover
      #debug
      show_status
   end_mover

   edit processor guidance
      #debug
      show_status
      show_diagnostics
      show_commands
      show_graphics
   end_processor
end_platform_type

platform launcher WSF_PLATFORM
   icon Wedge
   side blue

   add weapon rocket DELTA_IV_M
      launched_platform_type DELTA_IV_M_MOD
      launch_computer WSF_LAUNCH_COMPUTER
         commanded_loft_angle 88 deg       # <==== The desired flight path angle for the PITCH_OVER phase
      end_launch_computer
   end_weapon

   # Cape Canaveral, SLC-37 (Eastern Range Launch Site), inclinations 28-57
   position 28.531986n 80.566821w altitude 1 m agl heading 90 deg

   execute at_time 1 sec relative
      WeaponEntry(0).Fire();
   end_execute
end_platform

event_pipe
   file replay.aer
end_event_pipe

end_time 1200 sec
