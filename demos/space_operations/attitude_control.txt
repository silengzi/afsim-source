# =============================================================================
# A demonstration of generalized attitude control, which allows pointing
# the satellite or oriented platform part such as a sensor or comm device,
# along a particular vector to a target (e.g., another platform, nadir, the sun, etc).
# A 3D satellite model is used as a visual reference. A comm device, sensor,
# and  solar panel  (WSF_VISUAL_PART) are placed on an AFSIM platform
# in the same orientations as they appear on the 3D model.
# The solar panel is oriented with roughly a pitch up of 90 degrees.  T
# he comm device (dish antenna) points forward, along the model's x-axis.
# The sensor aperture is oriented with a pitch of -90 degrees.
# Try tethering to the satellite, then use pointing vectors in Mystic
# to the sun, ground station, and sensing target,
# to visualize that the satellite is pointing its sensor panels, comm device,
# or sensor, respectively.
#
# In order to do this, from the tethered view:
# - right click on the satellite
# - select "vectors"->"sun"
# - select "vectors"->"new vector"->"ground station"
# - select "vectors"->"new vector"->"target"
#
# In the main view, screen annotations will indicate when the satellite
# is pointing solar panels to the sun, antenna to ground station, or sensor to target.
# =============================================================================

antenna_pattern comm_antenna
   circular_pattern
      peak_gain  20 db
      beamwidth 0.5 deg
end_antenna_pattern

comm COMM WSF_RADIO_TRANSCEIVER
   transmitter
      power 1 kw
      frequency 1 ghz
      antenna_pattern comm_antenna
   end_transmitter
   receiver
   end_receiver
end_comm

platform_type ORIENTED_SATELLITE WSF_PLATFORM

   icon stereo_satellite

   mover WSF_SPACE_MOVER
      attitude_controller rate_limited
         maximum_yaw_rate 5 deg/s
         maximum_pitch_rate 5 deg/s
         maximum_roll_rate 5 deg/s
         orientation none
      end_attitude_controller
   end_mover

   // Orient the given articulated part along the vector from the platform to the target location, using a nadir z-axis constraint.
   script void OrientToTarget(WsfArticulatedPart aPart, Vec3 aTargetLocECI)
       Vec3 xAxis = Vec3.Construct(1.0, 0.0, 0.0);
       Vec3 xAxisECS = aPart.ConvertPCSVectorToECS(xAxis);
       Vec3 targetVecECI = Vec3.Subtract(aTargetLocECI, PLATFORM.LocationECI());
       Vec3 nadir = PLATFORM.LocationECI();
       nadir.Scale(-1.0);

       # Below is the call that enable generalized pointing:
       # Align "aPart" (PCS x-axis) with "targetVecECI".
       # Constrain "aPart" PCS z-axis to the nadir.
       ((WsfSpaceMover)PLATFORM.Mover()).SetOrientation(aPart, targetVecECI, nadir);

       //((WsfSpaceMover)PLATFORM.Mover()).SetOrientation(aPart.Yaw(), aPart.Pitch(), aPart.Roll(), targetVec, nadir); # alternate script method.

       # Draw a line indicating the pointing direction.
       WsfDraw draw = {};
       draw.Erase(0);
       draw.SetId(0);
       draw.SetColor(0.2, 0.8, 0.2);
       draw.SetLineSize(2);
       draw.BeginLines();
       draw.Vertex(aPart.Platform());
       draw.VertexWCS(PLATFORM.ConvertECIToWCS(aTargetLocECI));
       draw.End();
   end_script

   visual_part solar_panels WSF_VISUAL_PART
      pitch 90 degrees
   end_visual_part

   sensor sensor WSF_GEOMETRIC_SENSOR
      frame_time 2 s
      pitch -90 deg
      reports_location
      field_of_view circular
         half_angle 5 deg
      end_field_of_view
      yaw 0 degrees
   end_sensor

   comm comm COMM
      yaw 0 deg
   end_comm

   processor xmit WSF_SCRIPT_PROCESSOR
      update_interval 10 s
      on_update
         PLATFORM.Comm("comm").SendMessage(WsfControlMessage(), "ground_station", "comm");
      end_on_update
   end_processor

   processor Controller WSF_SCRIPT_PROCESSOR
      update_interval 10 s
      on_update
         WsfPlatform target = WsfSimulation.FindPlatform("target1");
         WsfPlatform groundStation = WsfSimulation.FindPlatform("ground_station");
         WsfSpaceMover sp = (WsfSpaceMover)PLATFORM.Mover();
         static double cMAX_RANGE = 1500000.0;
         if (PLATFORM.SlantRangeTo(target) < cMAX_RANGE)
         {
            writeln(TIME_NOW, " Orient sensor to target");
            OrientToTarget(PLATFORM.Sensor("sensor"), target.LocationECI());
         }
         else if (PLATFORM.SlantRangeTo(groundStation) < cMAX_RANGE)
         {
            writeln(TIME_NOW, "Orient comm to g station");
            OrientToTarget(PLATFORM.Comm("comm"), groundStation.LocationECI());
         }
         else
         {
            writeln(TIME_NOW, "Orient solar panels to sun");
            OrientToTarget(PLATFORM.VisualPart("solar_panels"), Sun.LocationECI(WsfDateTime.CurrentTime()));
         }
      end_on_update
   end_processor

end_platform_type

end_platform

script void ScreenPrint(string aName)
  WsfDraw draw = {};
  draw.SetId(1);
  draw.Erase(1);
  draw.SetTextSize(14);
  draw.SetColor(0.2, 1.0, 0.2);
  draw.BeginText(aName);
  draw.Vertex(WsfSimulation.FindPlatform("oriented_satellite1"));
  draw.End();
end_script
