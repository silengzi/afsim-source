script void DrawCovariance(WsfLocalTrack aLocalTrack)
   WsfCovariance cov = aLocalTrack.StateCovariance();
   if (cov.IsValid())
   {
      WsfDraw draw = {};
      draw.SetColor(1.0, 0.0, 0.0, 0.6);
      Ellipsoid ell = cov.EllipsoidNED();
      # The ellipsoid is enlarged to make it visible on typical visualization scales.
      draw.BeginEllipsoid(ell.OrientationHeading(),
                          ell.OrientationPitch(),
                          ell.OrientationRoll(),
                          10000 * ell.SemiAxisForward(),
                          10000 * ell.SemiAxisSide(),
                          10000 * ell.SemiAxisUp());
      draw.Vertex(aLocalTrack.ReportedLocation());
      draw.End();
   }
end_script

script void LocalTrackUpdated(WsfPlatform aPlatform, WsfLocalTrack aLocalTrack, WsfTrack aTrack)
   DrawCovariance(aLocalTrack);
end_script
