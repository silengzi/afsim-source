script void MoveCloudTracer(WsfPlatform aPlatform)
         lon += sLON_STEP;
         if (lon > 180.0)
         {
            lon = -180.0;
            lat -= sLAT_STEP;
            if (lat < -80.0)
            {
               aPlatform.Sensor("pen").TurnOff();
            }
         }
         aPlatform.SetLocation(lat, lon, alt);
end_script

script void SensorDetectionAttempt(WsfPlatform aPlatform, WsfSensor aSensor, WsfPlatform aTarget, WsfSensorInteraction aResult)
   if (aPlatform.Name() == "tracer")
   {
      bool detected = aResult.Detected();
      if (aTarget.Name() != "center")
      {
         writeln(aTarget.Name());
      }
      if (!detected)
      {
         WsfDraw draw = {};
         draw.SetColor(1.0, 1.0, 1.0, 0.4);
         draw.BeginQuadrilateral();
         draw.VertexLLA(lat + sLAT_STEP / 2.0, lon - sLON_STEP / 2.0, alt);
         draw.VertexLLA(lat + sLAT_STEP / 2.0, lon + sLON_STEP / 2.0, alt);
         draw.VertexLLA(lat - sLAT_STEP / 2.0, lon + sLON_STEP / 2.0, alt);
         draw.VertexLLA(lat - sLAT_STEP / 2.0, lon - sLON_STEP / 2.0, alt);
         draw.End();
      }
      MoveCloudTracer(aPlatform);
   }
end_script

observer
   enable SENSOR_DETECTION_ATTEMPT SensorDetectionAttempt
end_observer

end_time 1 hr
