  add comm sender1 WSF_RADIO_TRANSCEIVER

     on
     default_gateway
     network_name sender1

   transmitter
      antenna_pattern                     UNIFORM_0DBI_PATTERN
      attenuation                         none
      frequency                           1621 mhz  # ref 1
      bandwidth                           10.5 mhz  # ref 1
      propagation_model                   none
      polarization                        right_circular  # ref 1
      power                               50 dbw
   end_transmitter

   receiver
      antenna_pattern                     UNIFORM_0DBI_PATTERN
      frequency                           1621 mhz  # ref 1
      bandwidth                           10.5 mhz  # ref 1
      noise_power                         -105.0 dbm
      detection_threshold                 3.0 dB
   end_receiver

  end_comm


  add processor sender_router WSF_COMM_ROUTER
     interface sender1
  end_processor

   add sensor detect WSF_GEOMETRIC_SENSOR
      on
      reports_location
      frame_time 1 min
      azimuth_field_of_view -180 deg 180 deg
      elevation_field_of_view -90 deg 90 deg
   end_sensor

   // test script
   execute at_interval_of 1 s
      if (TIME_NOW > 10.0)
      {
         WsfControlMessage msg = WsfControlMessage();
         msg.SetFunction("blah blah blah");

#         PLATFORM.Comm("sender1").SendUnicastMessage(msg, commander1);
         WsfPlatform commander2 = WsfSimulation.FindPlatform("commander2");
         PLATFORM.Comm("sender1").SendUnicastMessage(msg, commander2);
      }
   end_execute

end_platform
