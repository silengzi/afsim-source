script void WriteNetwork(FileIO aFile)
   aFile.Writeln("network starlink_network WSF_COMM_NETWORK_GENERIC");
   aFile.Writeln("   comm_list");
   for (int plane = 0; plane < cmNumPlanes; plane += 1)
   {
      for (int sat = 0; sat < cmNumSatsPerPlane; sat += 1)
      {
         string name = GenerateName(plane, sat);
         aFile.Writeln("      member " + name + " crosslink1");         
         aFile.Writeln("      member " + name + " crosslink2");         
         aFile.Writeln("      member " + name + " crosslink3");         
         aFile.Writeln("      member " + name + " crosslink4");
         aFile.Writeln("      member " + name + " downlink");
         aFile.Writeln("");
      }
   }   
   aFile.Writeln("   end_comm_list");
   aFile.Writeln("end_network");
end_script
