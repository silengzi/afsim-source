# A constellation of sensing satellites monitors the build
# and reports images to a commander

processor REPORT WSF_SCRIPT_PROCESSOR
      on_message
         type WSF_VIDEO_MESSAGE
            script
               WsfVideoMessage message = (WsfVideoMessage)MESSAGE;
               if((message.Image().ObjectCount() == 0) ||
                  (!PLATFORM.CommEntry(0).PathExistsTo(WsfSimulation.FindPlatform("commander").Comm("commander").GetAddress())))
               {
                  SuppressMessage();
               }
            end_script
      end_on_message
end_processor

# Satellite constellation #####################################################
platform sat_sensor_1 SAT_SENSOR
   commander commander
   edit mover
      raan 0 deg
      anomaly 0 deg
   end_mover

   add comm sat_sensor_1 WSF_COMM_TRANSCEIVER
      network_name sat_sensor_1
   end_comm

   add sensor sat_sensor SAT_SENSOR
      internal_link report
   end_sensor

   add processor report REPORT
      report_to commander via sat_sensor_1 to commander
   end_processor

end_platform
