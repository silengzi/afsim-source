platform_type CHAFF WSF_PLATFORM
   on_initialize2
      # Make sure the spatial domain is properly defined Issue #90)
      string domain = WsfSimulation.FindPlatform("blue_airlifter_jet").SpatialDomain();
      if (SpatialDomain() != domain)
      {
         writeln("-FAIL- Platform: ", Name(), "; unexpected spatial domain: ", SpatialDomain(), ", expected: ", domain);
      }
   end_on_initialize2
end_platform_type

  #debug on
  terminal_velocity 1.0 m/s
  bloom_diameter 20 m
  expansion_time_constant 5.0 sec
  deceleration_rate 10 m/s2
  expiration_time 10 s

  number_dipoles 2000000

  frequency_maximum_rcs_table # if no units specified, assumed hz vs. dBsm
     independent_variable units mhz extrapolate
     100  71
     500  57
     1000 51
     1500 48
  end_frequency_maximum_rcs_table
end_chaff_parcel
