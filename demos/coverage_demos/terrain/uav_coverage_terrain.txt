// Load terrain data from DTED
terrain
   raw_dted 2 dted 39 -107 40 -106
end_terrain


# UTILITY - Used later to visually render the imported DTED data.
script void PlotRange(double startLat,   double startLon,
                      double endLat,     double endLon,
                      double latLonIncr, WsfDraw draw)

   for (double lat = startLat; lat < endLat; lat = lat + latLonIncr)
   {
      for (double lon = startLon; lon < endLon; lon = lon + latLonIncr)
      {
         WsfTerrain terrain = WsfTerrain();
         draw.VertexLLA(lat, lon, terrain.TerrainElevInterp(lat, lon));
         draw.VertexLLA(lat + latLonIncr, lon, terrain.TerrainElevInterp(lat + latLonIncr, lon));
         draw.VertexLLA(lat, lon, terrain.TerrainElevInterp(lat, lon));
         draw.VertexLLA(lat, lon + latLonIncr, terrain.TerrainElevInterp(lat, lon + latLonIncr));
      }
   }

end_script

# Set and save some variables for use during terrain drawing.
script_variables
   double gridLongitude = -114.95; # Longitude variable for use in drawing terrain grid.
   double gridHeight    = 3000.0;  # Height variable for use in drawing terrain grid.
end_script_variables

# Visually render the terrain as a 3-D mesh to display in Mystic.
script void draw_terrain()

   WsfDraw draw = WsfDraw();
   draw.BeginLines();

   WsfGeoPoint SW = WsfGeoPoint.Construct("39:00:00n 107:00:00w");
   WsfGeoPoint NE = WsfGeoPoint.Construct("40:00:00n 106:00:00w");    

   PlotRange(SW.Latitude(), SW.Longitude(), NE.Latitude(), NE.Longitude(), .005, draw);

   draw.End();
   draw.BeginLines();

   for (int i = 0; i < 100; i = i + 1)
   {
      draw.VertexLLA(36 - .05, gridLongitude, i * gridHeight / 99.0);
      draw.VertexLLA(36 + .05, gridLongitude, i * gridHeight / 99.0);
   }

   draw.End();

end_script

# One time only, render all terrain to display in Mystic.
execute at_time 1 s absolute
   draw_terrain();
end_execute
