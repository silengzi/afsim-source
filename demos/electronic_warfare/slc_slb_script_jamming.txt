# A simple EW radar (this somewhat corresponds to the EW/GCI radar defined
# in the Suppressor Obruty demo)

sensor EW_RADAR WSF_RADAR_SENSOR

   frame_time                10  sec
   # Specify the location and orientation of the sensor on the platform.
   location                  0.0 0.0 -30 m

   # The sensor sweeps only in azimuth
   scan_mode                 azimuth
   azimuth_scan_limits       -180 deg 180 deg

   # The field of view, minimum_range and maximum_range are not required.
   # They are simply used for quick culling and should be generous.
   # (i.e. the field_of_view limits should be probably be larger than the
   # corresponding slew limit because the beam can probably detect things
   # outside the slew limit).
   elevation_field_of_view  -2.5 deg 42.5 deg

   minimum_range        0 nm
   maximum_range      400 km  # really 225 km

   one_m2_detect_range 225 km

   transmitter
      power           750 kw
      frequency       900 mhz
      bandwidth       0.1 mhz
      pulse_width                  3  usec
      pulse_repetition_interval    750 usec // max PRI range of 225 km
      antenna_pattern EW_RADAR_ANTENNA
      antenna_tilt     10 deg
   end_transmitter

   receiver
      antenna_pattern EW_RADAR_ANTENNA
      antenna_tilt     10 deg
      frequency       900 mhz
      bandwidth         1 mhz

      electronic_protect
         debug
         technique sidelobe_blanking
         externally_controlled
         mitigated_technique_classes
            FALSE_TARGET_JAMMING
         end_mitigated_technique_classes

         effect slb_effect WSF_SLB_EFFECT
            debug
               antenna_pattern    AUX_ANTENNA
               blanking_threshold 3 dB
               saturation_effect
                  no_saturation_effect
                  duty_cycle_limit_effect
                  duty_cycle_limit 0.8
               end_saturation_effect
               target_blanking_effect
                  probabilities
                     pulse_density 0.0 0.0
                     pulse_density 0.1 0.2
                     pulse_density 0.3 0.6
                     pulse_density 0.5 0.8
                     pulse_density 1.0 1.0
                  end_probabilities
               end_target_blanking_effect
            end_effect
         end_technique

         technique sidelobe_cancelation
            externally_controlled
            mitigated_technique_classes
               NOISE_JAMMING
            end_mitigated_technique_classes

            effect slc_effect WSF_SLC_EFFECT
               debug
               number_canceller_channels  2
               cancellation_lock_ratio    3 dB
               antenna_pattern            AUX_ANTENNA
               cancellation_ratios
                  jammer_to_noise   0.0 db  0.0 db
                  jammer_to_noise  15.0 db 10.0 db
                  jammer_to_noise  30.0 db 20.0 db
                  saturation_ratio 10.0 dB
               end_cancellation_ratios
            end_effect
         end_technique
      end_electronic_protect
   end_receiver

   number_of_pulses_integrated 10

   jamming_perception_threshold 3 db

   hits_to_establish_track   3 5   # 1 of last 5 scans to establish track
   hits_to_maintain_track    1 4   # 1 of last 4 scans to maintain  track

   reports_location
   reports_velocity
   reports_signal_to_noise
   reports_range
   reports_bearing
   reports_elevation
end_sensor

processor SLB_SLC_Control WSF_SCRIPT_PROCESSOR
   #script_call_trace on
   script_debug_writes on

   update_interval 10.0 sec // update once per scan of the radar

   script_variables
      bool mSLB_Operating = false;
      bool mSLC_Operating = false;

      string mSensorName = "";
      string mSLB_TechName = "sidelobe_blanking";
      string mSLC_TechName = "sidelobe_cancelation";
   end_script_variables

   on_initialize
      if (mSensorName == "")
      {
         if (PLATFORM.SensorCount() == 0)
         {
            mSensorName = PLATFORM.SensorEntry(0).Name();
         }
      }

      if (mSensorName == "")
      {
         writeln("****WARNING: SLB_SLB_Operations script has null sensor name on ", PLATFORM.Name());
      }
   end_on_initialize

   on_update
      WsfSensor sensor = PLATFORM.Sensor(mSensorName);

      if (sensor.ContinuousJammingPerceived())
      {
         writeln_d("T=", TIME_NOW, " Continuous jamming perceived");
         if (! mSLC_Operating)
         {
            mSLC_Operating = sensor.SelectEP_Technique(mSLC_TechName);
         }

         if (mSLB_Operating && mSLC_Operating)
         {
            mSLB_Operating = (! sensor.DeselectEP_Technique(mSLB_TechName));
         }
      }
      else if (sensor.PulseJammingPerceived())
      {
         writeln_d("T=", TIME_NOW, " Pulse jamming perceived");
         if (! mSLB_Operating)
         {
            mSLB_Operating = sensor.SelectEP_Technique(mSLB_TechName);
         }

         if (mSLB_Operating && mSLC_Operating)
         {
            mSLC_Operating = (! sensor.DeselectEP_Technique(mSLC_TechName));
         }
      }
      else
      {
         writeln_d("T=", TIME_NOW, " Jamming not perceived");
         if (mSLC_Operating)
         {
            mSLC_Operating = (! sensor.DeselectEP_Technique(mSLC_TechName));
         }

         if (mSLB_Operating)
         {
            mSLB_Operating = (! sensor.DeselectEP_Technique(mSLB_TechName));
         }
      }
      writeln("   SLB=", mSLB_Operating);
      writeln("   SLC=", mSLC_Operating);
   end_on_update
end_processor
